title: Banner Ads
description: This guide explains how to integrate banner ads into a React Native app.

Banner ad units display rectangular ads that occupy a portion of an app’s layout. They can refresh automatically on a set interval and are the simplest ad format to implement.

Banner ads in React Native are handled via the `BannerAdView` component from **react-native-cas**, used directly in JSX.

CAS provides an `autoload` option to simplify app logic. Decide whether to enable `autoload` or to manage load cycles manually.

```ts
const isAutoReloadEnabled = true as const;
```

Manual mode (`autoload = false`) is useful when you show a single ad per screen and screens change frequently. `refreshInterval` works independently of `autoload`.

## Get the Ad size
Use the `BannerAdSize` enum:

```ts
const size = BannerAdSize.ADAPTIVE;   // Flexible width, base height ~50
const size2 = BannerAdSize.SMART;     // ~320x50 on phones
const size3 = BannerAdSize.MREC;      // 300x250
const size4 = BannerAdSize.LEADERBOARD; // 728x90 (tablet only)
const size5 = BannerAdSize.BANNER;    // 320x50
```

## Create Ad instance
```tsx
import React from 'react';
import { View } from 'react-native';
import { BannerAdView, BannerAdSize } from 'react-native-cas';

export default function BannerAdExample() {
  return (
    <View style={{ alignItems: 'center' }}>
      <BannerAdView
        size={BannerAdSize.ADAPTIVE}
        autoload={true}
        refreshInterval={30}
        onAdViewLoaded={({ width, height }) =>
          console.log(`Banner loaded: ${width}x${height}`)
        }
        onAdViewFailed={({ code, message }) =>
          console.log(`Banner failed: ${code} ${message}`)
        }
        onAdViewImpression={(impression) =>
          console.log('Impression:', impression)
        }
        onAdViewClicked={() => console.log('Banner clicked')}
      />
    </View>
  );
}
```

Keep the view mounted to avoid delays. With **Adaptive** banners, height adjusts automatically on width changes (e.g., orientation) — no manual reload needed.

## Ad refresh interval
```tsx
<BannerAdView
  size={BannerAdSize.ADAPTIVE}
  refreshInterval={30}
/>
```

## Ad content callbacks
```tsx
<BannerAdView
  size={BannerAdSize.BANNER}
  onAdViewLoaded={() => console.log('Ad loaded')}
  onAdViewFailed={(e) => console.log('Failed to load:', e.message)}
  onAdViewImpression={(info) => console.log('Impression:', info)}
  onAdViewClicked={() => console.log('Ad clicked')}
/>
```

## Show Ad View
```tsx
<View style={{ flex: 1, justifyContent: 'flex-end', alignItems: 'center' }}>
  <BannerAdView size={BannerAdSize.ADAPTIVE} />
</View>
```

While loading, the view is minimal (1×1). It expands automatically when the ad loads. Ensure the container is large enough; padding reduces available space.

## Manual reload Ad
```tsx
import React, { useRef } from 'react';
import { Button } from 'react-native';
import { BannerAdView, BannerAdSize, type BannerAdViewRef } from 'react-native-cas';

export default function ReloadExample() {
  const ref = useRef<BannerAdViewRef>(null);
  return (
    <>
      <BannerAdView ref={ref} size={BannerAdSize.BANNER} />
      <Button title="Reload Ad" onPress={() => ref.current?.loadAd()} />
    </>
  );
}
```

### BannerAdView props
<Property name="size" type="BannerAdSize" required>
  Banner size (`BANNER`, `ADAPTIVE`, `MREC`, `LEADERBOARD`, `SMART`).
</Property>

---

<Property name="autoload" type="boolean" optional>
  Automatically loads the next creative after a show/failure.
</Property>

---

<Property name="refreshInterval" type="number" optional>
  Auto-refresh interval in seconds. `0` disables refreshing.
</Property>

---

<Property name="maxWidth" type="number" optional>
  Max width used for adaptive banners. Defaults to device width.
</Property>

---

<Property name="maxHeight" type="number" optional>
  Max height for inline adaptive banners. Defaults to device height.
</Property>

---

<Property name="casId" type="string" optional>
  Optional CAS ID override for this specific view.
</Property>

---

<Property name="onAdViewLoaded" type="(info: AdViewInfo) => void" optional>
  Called when the ad is successfully loaded (contains actual width/height).
</Property>

---

<Property name="onAdViewFailed" type="(error: AdError) => void" optional>
  Called when loading fails.
</Property>

---

<Property name="onAdViewClicked" type="() => void" optional>
  Called when the ad is clicked.
</Property>

---

<Property name="onAdViewImpression" type="(info: AdContentInfo) => void" optional>
  Called when an impression is recorded with monetization info.
</Property>
